
Функция ПолучитьЭлементЧекЛистаПоСтроке(пСтрока, пСсылка) Экспорт 
	ссылка = Справочники.ПунктЧекЛиста.НайтиПоНаименованию(пСтрока, , , пСсылка);
	Если ссылка.Пустая() Тогда
		новЭлемент = Справочники.ПунктЧекЛиста.СоздатьЭлемент();
		новЭлемент.Владелец = пСсылка;
		новЭлемент.Наименование = пСтрока;
		новЭлемент.Записать();
		
		ссылка = новЭлемент.Ссылка;
	КонецЕсли;
	
	Возврат ссылка;
КонецФункции // ПолучитьЭлементЧекЛистаПоСтроке()
